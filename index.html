            // ── Подсчёт баллов ──────────────────────────────

            function calculateScores() {

                const scores = { 1: 0, 2: 0, 3: 0 };

                const totalPerLevel = { 1: 4, 2: 4, 3: 4 };

                allCheckboxes.forEach(cb => {

                    const level = parseInt(cb.getAttribute('data-level'));

                    if (cb.checked && scores.hasOwnProperty(level)) {

                        scores[level]++;

                    }

                });

                return { scores, totalPerLevel };

            }

            // ── Определение дефицитных уровней ──────────────

            function findDeficitLevels(scores) {

                const minScore = Math.min(scores[1], scores[2], scores[3]);

                const deficitLevels = [];

                for (const lvl of [1, 2, 3]) {

                    if (scores[lvl] === minScore) {

                        deficitLevels.push(lvl);

                    }

                }

                // Если все уровни набрали 4/4 — всё равно показываем все для информации

                if (minScore === 4 && deficitLevels.length === 3) {

                    return { levels: [1, 2, 3], isAllStrong: true };

                }

                return { levels: deficitLevels, isAllStrong: minScore === 4 };

            }

            // ── Отрисовка результатов ───────────────────────

            function renderResults(deficitLevels, scores, isAllStrong) {

                // Цитата

                if (isAllStrong && deficitLevels.length === 3) {

                    resultsQuote.textContent =

                        'Ваша экосистема в прекрасном балансе! Вот инструменты для дальнейшего роста.';

                } else if (deficitLevels.length === 1) {

                    resultsQuote.textContent = toolsDB[deficitLevels[0]].inspiration;

                } else {

                    const quotes = deficitLevels.map(l => toolsDB[l].inspiration);

                    resultsQuote.textContent = quotes.join(' · ');

                }

                // Тег уровня

                const levelTags = deficitLevels.map(l => levelNames[l]);

                resultsLevelTag.innerHTML = levelTags

                    .map(t => `<span>${t}</span>`)

                    .join(' ');

                // Инструменты (собираем уникальные по URL для объединённых результатов)

                const seenUrls = new Set();

                const allTools = [];

                for (const lvl of deficitLevels) {

                    for (const tool of toolsDB[lvl].items) {

                        if (!seenUrls.has(tool.url)) {

                            seenUrls.add(tool.url);

                            allTools.push(tool);

                        }

                    }

                }

                resultsTools.innerHTML = allTools

                    .map(

                        tool => `

                    <div class="tool-card">

                        <div>

                            <p class="tool-card__name">${tool.name}</p>

                            <p class="tool-card__desc">${tool.desc}</p>

                        </div>

                        <a href="${tool.url}" target="_blank" rel="noopener noreferrer" class="btn btn--outline btn--small">Перейти →</a>

                    </div>

                `

                    )

                    .join('');

                // Скролл к результатам

                }, 150);

            function renderScoreBadges(scores) {

                const badgeClass = (score) => {

                    if (score === 4) return 'score-badge--strong';

                    if (score >= 2) return 'score-badge--moderate';

                    return 'score-badge--deficit';

                };

                const badgeLabel = (score, level) => {

                    const shortName = { 1: 'Фундамент', 2: 'Среда', 3: 'Крона' };

                    return `${shortName[level]}: ${score}/4`;

                };

                scoreBadges.innerHTML = [1, 2, 3]

                    .map(

                        l =>

                        `<span class="score-badge ${badgeClass(scores[l])}">${badgeLabel(scores[l], l)}</span>`

                    )

                    .join('');

                scoreBadges.style.display = 'flex';

            }
